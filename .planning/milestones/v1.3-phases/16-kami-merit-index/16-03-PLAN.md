    def test_merit_scores_in_audit_hash():
        """merit_scores must appear in MiFID II audit hash (NOT in AUDIT_EXCLUDED_FIELDS).

        Two state snapshots with identical non-merit fields but different merit_scores
        must produce different hashes. Same merit_scores with rounded floats must
        produce the same hash (4dp rounding prevents float jitter).
        """
        from src.core.audit_logger import AuditLogger, AUDIT_EXCLUDED_FIELDS
        from src.core.kami import DEFAULT_MERIT

        assert "merit_scores" not in AUDIT_EXCLUDED_FIELDS, (
            "merit_scores must NOT be excluded from audit hash — it is numeric decision context"
        )

        logger = AuditLogger()

        state_a = {
            "task_id": "test-001",
            "node_id": "debate_synthesizer",
            "merit_scores": {"AXIOM": {"composite": round(0.7312, 4)}},
        }
        state_b = {
            "task_id": "test-001",
            "node_id": "debate_synthesizer",
